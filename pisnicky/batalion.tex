\nazev{Batalion}{Spiritual kvintet}

\slokaB{R1:}{ 
	Víno máš a markytánku, dlouhá noc se prohýří,\n víno máš a chvilku spánku, díky, díky, verbíři.
	}

\slokaB{1.}{
	Dříve, než se rozední, kapitán k osedlání rozkaz dává,\n ostruhami do slabin koně pohání,
	Tam na straně polední čekají ženy, zlaťáky a sláva,\n do výstřelů z karabin zvon už vyzvání.	
	}	

\slokaB{R2:}{
	Víno na kuráž a pomilovat markytánku,\n zítra do Burgund batalion zamíří,\n
	Víno na kuráž a k ránu dvě hodiny spánku,\n díky, díky vám, královští verbíři.
	} 

\slokaB{2.}{
	Rozprášen je batalion, poslední vojáci se k zemi hroutí,\n na polštáři z kopretin budou věčně spát,\n
	Neplač, sladká Marion, verbíři nové chlapce přivedou ti,\n za královský hermelín padne každý rád.	
	}
