\slokaB{2.}{
	Vítr si tiše hvízdá,\n
	po silnici spálený.\n
	V tom městě nikdo nezdraví.\n
	Šerif i soudce -- gangsteři, oba řádně zvolený,\n
	a lidi strachem nezdraví.
	}

\slokaB{3.}{
	Sto cizejch zabíječů s pistolema skotačí\n
	a zákon džungle panuje, je, je, je.\n
	Provazník plete smyčky,\n
	hrobař kopat nestačí\n
	A truhlář rakve hobluje.
	}
